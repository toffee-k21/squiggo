model Sketch {
  id      Int    @id @default(autoincrement())
  type    String
  roomId  Int
  message String
  userId  String
  room    Room   @relation(fields: [roomId], references: [id])
  user    User   @relation(fields: [userId], references: [id])
}

model Chat {
  id      Int    @id @default(autoincrement())
  type    String
  roomId  Int
  message String
  userId  String
  room    Room   @relation(fields: [roomId], references: [id])
  user    User   @relation(fields: [userId], references: [id])
}
